unsigned short csum (unsigned short *packet, int packlen) {
 register unsigned long sum = 0;
 while (packlen > 1) {
  sum+= *(packet++);
  packlen-=2;
 }
 if (packlen > 0)
  sum += *(unsigned char *)packet;
 while (sum >> 16)
  sum = (sum & 0xffff) + (sum >> 16);

 return (unsigned short) ~sum;
}
